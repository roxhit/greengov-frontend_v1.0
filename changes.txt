import { useState, useEffect } from “react”;
import {
allocateResource,
updateResource,
deleteResource,
updateResourceStatus,
getAllResources,
createInfrastructure,
getAllInfrastructure,
deleteInfrastructure,
} from “../api/resourceApi”;

import Loading from “../components/Loading”;
import Alert from “../components/Alert”;
import { RequiredPermission } from “../components/PermissionGate”;

export default function ResourcesPage() {
const [activeTab, setActiveTab] = useState(“resources”);
const [resources, setResources] = useState([]);
const [infrastructure, setInfrastructure] = useState([]);
const [selected, setSelected] = useState({});
const [mode, setMode] = useState(””);
const [section, setSection] = useState(””);
const [showModal, setShowModal] = useState(false);
const [loading, setLoading] = useState(false);
const [error, setError] = useState(””);
const [success, setSuccess] = useState(””);

const clearMessages = () => { setError(””); setSuccess(””); };

useEffect(() => {
if (activeTab === “infrastructure”) loadInfra();
}, [activeTab]);

useEffect(() => {
if (activeTab === “resources”) loadResources();
}, [activeTab]);

const loadResources = async () => {
try {
setLoading(true);
const data = await getAllResources();
setResources(Array.isArray(data) ? data : []);
} catch {
setError(“Failed to load resources”);
} finally {
setLoading(false);
}
};

const loadInfra = async () => {
try {
setLoading(true);
const data = await getAllInfrastructure();
setInfrastructure(Array.isArray(data) ? data : []);
} catch {
setError(“Failed to load infrastructure”);
} finally {
setLoading(false);
}
};

const openModal = (item, mode, section) => {
setSelected(item || {});
setMode(mode);
setSection(section);
setShowModal(true);
};

const handleSave = async () => {
try {
setLoading(true);
clearMessages();
if (section === “resource”) {
if (mode === “add”) {
await allocateResource(selected);
setResources([…resources, selected]);
setSuccess(“Resource added ✅”);
} else if (mode === “edit”) {
await updateResource(selected.id, selected);
setResources(resources.map(r => r.id === selected.id ? selected : r));
setSuccess(“Resource updated ✅”);
}
} else {
if (mode === “add”) {
await createInfrastructure(selected);
await loadInfra();
setSuccess(“Infrastructure added ✅”);
}
}
} catch {
setError(“Operation failed ❌”);
} finally {
setLoading(false);
setShowModal(false);
}
};

const handleDelete = async () => {
try {
setLoading(true);
if (section === “resource”) {
await deleteResource(selected.id);
setResources(resources.filter(r => r.id !== selected.id));
} else {
await deleteInfrastructure(selected.id);
await loadInfra();
}
setSuccess(“Deleted successfully ✅”);
} catch {
setError(“Delete failed ❌”);
} finally {
setLoading(false);
setShowModal(false);
}
};

const handleStatus = async (item) => {
try {
await updateResourceStatus(item.id, “Updated”);
setSuccess(“Status updated ✅”);
} catch {
setError(“Status update failed ❌”);
}
};

const badgeStyle = {
fontSize: “0.7rem”,
letterSpacing: “0.05em”,
padding: “0.35em 0.65em”,
};

return (
<RequiredPermission authority="PROGRAM_MANAGER">
<style>{`.resource-table thead th { background-color: #1a1f2e; color: #6c757d; font-size: 0.7rem; font-weight: 600; letter-spacing: 0.1em; text-transform: uppercase; border-bottom: 1px solid #2d3447; padding: 0.85rem 1rem; } .resource-table tbody tr { border-bottom: 1px solid #1e2435; transition: background-color 0.15s ease; } .resource-table tbody tr:hover { background-color: #1e2435 !important; } .resource-table tbody td { padding: 0.85rem 1rem; vertical-align: middle; font-size: 0.875rem; color: #c9d1e0; border: none; } .resource-table .id-cell { font-family: monospace; font-size: 0.75rem; color: #4b5675; } .tab-pill { border: none; background: transparent; color: #6c757d; font-size: 0.82rem; font-weight: 500; letter-spacing: 0.04em; padding: 0.5rem 1.1rem; border-radius: 20px; transition: all 0.2s ease; } .tab-pill.active { background-color: #198754; color: #fff; } .tab-pill:hover:not(.active) { background-color: #1e2435; color: #adb5bd; } .action-btn { font-size: 0.72rem; font-weight: 500; letter-spacing: 0.03em; padding: 0.3rem 0.65rem; border-radius: 6px; } .table-card { background-color: #141824; border: 1px solid #1e2435; border-radius: 12px; overflow: hidden; } .modal-content.styled { background-color: #141824; border: 1px solid #2d3447; border-radius: 14px; } .modal-content.styled .modal-header { border-bottom: 1px solid #1e2435; padding: 1.2rem 1.5rem; } .modal-content.styled .modal-footer { border-top: 1px solid #1e2435; padding: 1rem 1.5rem; } .styled-input { background-color: #1a1f2e; border: 1px solid #2d3447; color: #c9d1e0; border-radius: 8px; font-size: 0.875rem; padding: 0.55rem 0.85rem; transition: border-color 0.15s ease; } .styled-input:focus { background-color: #1e2435; border-color: #198754; color: #e0e6f0; box-shadow: 0 0 0 3px rgba(25, 135, 84, 0.15); outline: none; } .styled-input::placeholder { color: #4b5675; } .empty-state { padding: 3rem 1rem; text-align: center; color: #4b5675; font-size: 0.875rem; }`}</style>

```
  <div>
    {/* Header */}
    <div className="d-flex align-items-center justify-content-between mb-4">
      <div>
        <h5 className="fw-semibold text-white mb-0">Resource & Infrastructure</h5>
        <p className="text-secondary mb-0" style={{ fontSize: "0.8rem" }}>
          Manage your allocated resources and infrastructure
        </p>
      </div>
    </div>

    {/* Alerts */}
    {error && <Alert message={error} type="danger" />}
    {success && <Alert message={success} type="success" />}
    {loading && <Loading />}

    {/* Tabs */}
    <div className="d-flex gap-1 mb-4 p-1" style={{ background: "#0f1118", borderRadius: "24px", display: "inline-flex" }}>
      <button
        className={`tab-pill ${activeTab === "resources" ? "active" : ""}`}
        onClick={() => setActiveTab("resources")}
      >
        Resources
      </button>
      <button
        className={`tab-pill ${activeTab === "infrastructure" ? "active" : ""}`}
        onClick={() => setActiveTab("infrastructure")}
      >
        Infrastructure
      </button>
    </div>

    {/* Resources Table */}
    {activeTab === "resources" && (
      <div>
        <div className="d-flex justify-content-end mb-3">
          <button
            className="btn btn-success btn-sm px-3"
            style={{ borderRadius: "8px", fontSize: "0.82rem", fontWeight: 500 }}
            onClick={() => openModal({}, "add", "resource")}
          >
            + Add Resource
          </button>
        </div>

        <div className="table-card">
          <table className="table resource-table mb-0">
            <thead>
              <tr>
                <th>ID</th>
                <th>Project</th>
                <th>Type</th>
                <th>Quantity</th>
                <th>Actions</th>
              </tr>
            </thead>
            <tbody>
              {resources.length === 0 ? (
                <tr>
                  <td colSpan={5}>
                    <div className="empty-state">No resources found</div>
                  </td>
                </tr>
              ) : resources.map(r => (
                <tr key={r.id}>
                  <td className="id-cell">#{r.id}</td>
                  <td className="fw-medium">{r.projectId}</td>
                  <td>
                    <span className="badge bg-success bg-opacity-10 text-success" style={badgeStyle}>
                      {r.type}
                    </span>
                  </td>
                  <td>{r.quantity}</td>
                  <td>
                    <div className="d-flex gap-1">
                      <button className="btn btn-sm action-btn btn-outline-secondary"
                        onClick={() => openModal(r, "view", "resource")}>
                        View
                      </button>
                      <button className="btn btn-sm action-btn btn-outline-warning"
                        onClick={() => openModal(r, "edit", "resource")}>
                        Edit
                      </button>
                      <button className="btn btn-sm action-btn btn-outline-info"
                        onClick={() => handleStatus(r)}>
                        Status
                      </button>
                      <button className="btn btn-sm action-btn btn-outline-danger"
                        onClick={() => openModal(r, "delete", "resource")}>
                        Delete
                      </button>
                    </div>
                  </td>
                </tr>
              ))}
            </tbody>
          </table>
        </div>
      </div>
    )}

    {/* Infrastructure Table */}
    {activeTab === "infrastructure" && (
      <div>
        <div className="d-flex justify-content-end mb-3">
          <button
            className="btn btn-success btn-sm px-3"
            style={{ borderRadius: "8px", fontSize: "0.82rem", fontWeight: 500 }}
            onClick={() => openModal({}, "add", "infra")}
          >
            + Add Infrastructure
          </button>
        </div>

        <div className="table-card">
          <table className="table resource-table mb-0">
            <thead>
              <tr>
                <th>ID</th>
                <th>Type</th>
                <th>Location</th>
                <th>Capacity</th>
                <th>Actions</th>
              </tr>
            </thead>
            <tbody>
              {infrastructure.length === 0 ? (
                <tr>
                  <td colSpan={5}>
                    <div className="empty-state">No infrastructure found</div>
                  </td>
                </tr>
              ) : infrastructure.map(i => (
                <tr key={i.id}>
                  <td className="id-cell">#{i.id}</td>
                  <td>
                    <span className="badge bg-primary bg-opacity-10 text-primary" style={badgeStyle}>
                      {i.type}
                    </span>
                  </td>
                  <td>{i.location}</td>
                  <td>{i.capacity}</td>
                  <td>
                    <div className="d-flex gap-1">
                      <button className="btn btn-sm action-btn btn-outline-secondary"
                        onClick={() => openModal(i, "view", "infra")}>
                        View
                      </button>
                      <button className="btn btn-sm action-btn btn-outline-danger"
                        onClick={() => openModal(i, "delete", "infra")}>
                        Delete
                      </button>
                    </div>
                  </td>
                </tr>
              ))}
            </tbody>
          </table>
        </div>
      </div>
    )}

    {/* Modal */}
    {showModal && (
      <>
        <div className="modal-backdrop fade show" style={{ backdropFilter: "blur(2px)" }}></div>
        <div className="modal d-block">
          <div className="modal-dialog modal-dialog-centered">
            <div className="modal-content styled text-white">

              <div className="modal-header">
                <div>
                  <h6 className="fw-semibold text-white mb-0 text-capitalize">
                    {mode === "add" ? "New" : mode === "edit" ? "Edit" : mode === "delete" ? "Confirm Delete" : "View"}{" "}
                    {section === "infra" ? "Infrastructure" : "Resource"}
                  </h6>
                  {mode === "delete" && (
                    <p className="text-secondary mb-0" style={{ fontSize: "0.75rem" }}>
                      This action cannot be undone
                    </p>
                  )}
                </div>
                <button className="btn-close btn-close-white" onClick={() => setShowModal(false)}></button>
              </div>

              <div className="modal-body px-4 py-3">
                {mode === "delete" ? (
                  <p className="text-secondary mb-0">
                    Are you sure you want to delete this {section === "infra" ? "infrastructure" : "resource"}?
                  </p>
                ) : (
                  <div className="d-flex flex-column gap-2">
                    <input
                      className="form-control styled-input"
                      placeholder="Project ID"
                      value={selected.projectId || ""}
                      disabled={mode === "view"}
                      onChange={(e) => setSelected({ ...selected, projectId: e.target.value })}
                    />
                    <input
                      className="form-control styled-input"
                      placeholder="Type"
                      value={selected.type || ""}
                      disabled={mode === "view"}
                      onChange={(e) => setSelected({ ...selected, type: e.target.value })}
                    />
                    {section === "resource" && (
                      <input
                        className="form-control styled-input"
                        placeholder="Quantity"
                        value={selected.quantity || ""}
                        disabled={mode === "view"}
                        onChange={(e) => setSelected({ ...selected, quantity: e.target.value })}
                      />
                    )}
                    {section === "infra" && (
                      <>
                        <input
                          className="form-control styled-input"
                          placeholder="Location"
                          value={selected.location || ""}
                          disabled={mode === "view"}
                          onChange={(e) => setSelected({ ...selected, location: e.target.value })}
                        />
                        <input
                          className="form-control styled-input"
                          placeholder="Capacity"
                          value={selected.capacity || ""}
                          disabled={mode === "view"}
                          onChange={(e) => setSelected({ ...selected, capacity: e.target.value })}
                        />
                      </>
                    )}
                  </div>
                )}
              </div>

              <div className="modal-footer">
                <button
                  className="btn btn-sm btn-outline-secondary px-3"
                  style={{ borderRadius: "8px" }}
                  onClick={() => setShowModal(false)}
                >
                  Cancel
                </button>
                {mode === "delete" ? (
                  <button
                    className="btn btn-sm btn-danger px-3"
                    style={{ borderRadius: "8px" }}
                    onClick={handleDelete}
                  >
                    Delete
                  </button>
                ) : mode !== "view" && (
                  <button
                    className="btn btn-sm btn-success px-3"
                    style={{ borderRadius: "8px" }}
                    onClick={handleSave}
                  >
                    Save Changes
                  </button>
                )}
              </div>

            </div>
          </div>
        </div>
      </>
    )}
  </div>
</RequiredPermission>
```

);
}
